CREATE TABLE CUSTOMER(
	ControlL_ID INT(3) AUTO_INCREMENT PRIMARY KEY,
	User_ID VARCHAR(10),
	MailAddress VARCHAR(50),
	Password VARCHAR(15),
	PhoneNo VARCHAR(10),
	PostalCode VARCHAR(7),
	Address VARCHAR(150),
	Name VARCHAR(10)
);

CREATE TABLE GOODS(
	Goods_ID INT(3) AUTO_INCREMENT PRIMARY KEY,
	Price INT(5),
	Category_ID INT(3) 
);

CREATE TABLE GOODS_CATEGORY(
	Category_ID INT(3) AUTO_INCREMENT PRIMARY KEY,
	Category_Name VARCHAR(10)
);

CREATE TABLE GOODS_DETAILS(
	Goods_ID INT(3) PRIMARY KEY,
	Goods_Name VARCHAR(20),
	Goods_Text VARCHAR(200)
);

CREATE TABLE CART(
	Goods_ID INT(3) PRIMARY KEY,
	Quantity INT(3)
);

ALTER TABLE GOODS ADD FOREIGN KEY(Category_ID) REFERENCES GOODS_CATEGORY(Category_ID);

ALTER TABLE GOODS_DETAILS ADD FOREIGN KEY(Goods_ID) REFERENCES GOODS(Goods_ID);

ALTER TABLE CART ADD FOREIGN KEY(Goods_ID) REFERENCES GOODS(Goods_ID);
